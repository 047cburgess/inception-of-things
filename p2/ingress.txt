kubectl create deployment hello-world --image=nginx

kubectl expose deployment hello-world --port=80 --type=NodePort

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: myapp-ingress
spec:
  rules:
    - host: app1.com
      http:
        paths:
          - pathType: Prefix
            path: "/"
            backend:
              service:
                name: hello-world
                port:
                  number: 80


kubectl apply -f ingress.yml
